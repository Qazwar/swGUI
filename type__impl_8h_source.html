389 template<typename F>
390 RTTR_INLINE void type::register_converter_func(F func)
391 {
392  using namespace detail;
393 
394  using target_type_orig = typename function_traits<F>::return_type;
395  using target_type = remove_cv_t<remove_reference_t<target_type_orig>>;
396 
397  const std::size_t arg_count = function_traits<F>::arg_count;
398 
399  static_assert(arg_count == 2, "Invalid argument count! The converter function signature must be: <target_type(source_type, bool&)>");
400  static_assert(!std::is_same<void, target_type>::value, "Return type cannot be void!");
401  static_assert(std::is_same<bool&, typename param_types<F, 1>::type>::value, "Second argument type must be a bool reference(bool&).");
402 
403  using source_type_orig = param_types_t<F, 0>;
404  using source_type = remove_cv_t<remove_reference_t<source_type_orig>>;
405 
406  auto converter = detail::make_unique<type_converter<target_type, source_type, F>>(func);
407  type source_t = type::get<source_type>();
408  type_register::converter(source_t, std::move(converter));
409 }
