218 template<typename T, typename Tp, typename Converter>
219 struct variant_data_base_policy
220 {
221  static bool invoke(variant_policy_operation op, const variant_data& src_data, argument_wrapper arg)
222  {
223  switch (op)
224  {
225  case variant_policy_operation::DESTROY:
226  {
227  Tp::destroy(const_cast<T&>(Tp::get_value(src_data)));
228  break;
229  }
230  case variant_policy_operation::CLONE:
231  {
232  Tp::clone(Tp::get_value(src_data), arg.get_value<variant_data>());
233  break;
234  }
235  case variant_policy_operation::SWAP:
236  {
237  Tp::swap(const_cast<T&>(Tp::get_value(src_data)), arg.get_value<variant_data>());
238  break;
239  }
240  case variant_policy_operation::EXTRACT_WRAPPED_VALUE:
241  {
242  arg.get_value<variant>() = get_wrapped_value(Tp::get_value(src_data));
243  break;
244  }
245  case variant_policy_operation::GET_VALUE:
246  {
247  arg.get_value<const void*>() = &Tp::get_value(src_data);
248  break;
249  }
250  case variant_policy_operation::GET_TYPE:
251  {
252  arg.get_value<type>() = type::get<T>();
253  break;
254  }
255  case variant_policy_operation::GET_PTR:
256  {
257  arg.get_value<void*>() = as_void_ptr(std::addressof(Tp::get_value(src_data)));
258  break;
259  }
260  case variant_policy_operation::GET_RAW_TYPE:
261  {
262  arg.get_value<type>() = type::get<typename raw_type<T>::type>();
263  break;
264  }
265  case variant_policy_operation::GET_RAW_PTR:
266  {
267  arg.get_value<void*>() = as_void_ptr(raw_addressof(Tp::get_value(src_data)));
268  break;
269  }
270  case variant_policy_operation::GET_ADDRESS_CONTAINER:
271  {
272  data_address_container& data = arg.get_value<data_address_container>();
273 
274  data.m_type = type::get< raw_addressof_return_type_t<T> >();
275  data.m_wrapped_type = type::get< wrapper_address_return_type_t<T> >();
276  data.m_data_address = as_void_ptr(raw_addressof(Tp::get_value(src_data)));
277  data.m_data_address_wrapped_type = as_void_ptr(wrapped_raw_addressof(Tp::get_value(src_data)));
278  break;
279  }
280  case variant_policy_operation::IS_ARRAY:
281  {
282  return can_create_array_container<T>::value;
283  }
284  case variant_policy_operation::TO_ARRAY:
285  {
286  arg.get_value<std::unique_ptr<array_wrapper_base>&>() = create_variant_array_view(const_cast<T&>(Tp::get_value(src_data)));
287  break;
288  }
289  case variant_policy_operation::CONVERT:
290  {
291  return Converter::convert_to(Tp::get_value(src_data), arg.get_value<argument>());
292  break;
293  }
294  case variant_policy_operation::IS_VALID:
295  {
296  return true;
297  break;
298  }
299  case variant_policy_operation::IS_NULLPTR:
300  {
301  return is_nullptr(Tp::get_value(src_data));
302  break;
303  }
304  case variant_policy_operation::COMPARE_EQUAL:
305  {
306  const auto& param = arg.get_value<std::tuple<const variant&, const variant&>>();
307  const variant& lhs = std::get<0>(param);
308  const variant& rhs = std::get<1>(param);
309  const type rhs_type = rhs.get_type();
310  const type lhs_type = type::get<T>();
311  const bool is_lhs_arithmetic = std::is_arithmetic<T>::value;
312 
313  if (lhs_type == rhs_type)
314  {
315  return COMPARE_EQUAL_PRE_PROC(src_data, rhs);
316  }
317  else
318  {
319  if (is_lhs_arithmetic && rhs_type.is_arithmetic())
320  {
321  return variant_compare_equal(lhs, lhs_type, rhs, rhs_type);
322  }
323  else
324  {
325  variant var_tmp;
326  if (rhs.convert(lhs_type, var_tmp))
327  return COMPARE_EQUAL_PRE_PROC(src_data, var_tmp);
328  else if (lhs.convert(rhs_type, var_tmp))
329  return (var_tmp.compare_equal(rhs));
330  }
331  }
332 
333  return false;
334  break;
335  }
336  case variant_policy_operation::COMPARE_LESS:
337  {
338  const auto& param = arg.get_value<std::tuple<const variant&, const variant&>>();
339  const variant& lhs = std::get<0>(param);
340  const variant& rhs = std::get<1>(param);
341  const type rhs_type = rhs.get_type();
342  const type lhs_type = type::get<T>();
343  int result = 0;
344  if (lhs_type == rhs_type)
345  {
346  if (COMPARE_LESS_PRE_PROC(src_data, rhs, result))
347  return (result == -1 ? true : false);
348  }
349  else
350  {
351  return variant_compare_less(lhs, lhs_type, rhs, rhs_type);
352  }
353 
354  // as last try, do a string conversion
355  return (lhs.to_string() < rhs.to_string());
356  break;
357  }
358  }
359 
360  return true;
361  }
362 };

876  {

878  }

901  }

919  }
