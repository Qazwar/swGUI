618 template<typename CharT, typename Traits>
619 inline void insert_fill_chars(std::basic_ostream<CharT, Traits>& os, std::size_t n)
620 {
621  enum { chunk_size = 8 };
622  CharT fill_chars[chunk_size];
623  std::fill_n(fill_chars, static_cast< std::size_t >(chunk_size), os.fill());
624  for (; n >= chunk_size && os.good(); n -= chunk_size)
625  os.write(fill_chars, static_cast< std::size_t >(chunk_size));
626  if (n > 0 && os.good())
627  os.write(fill_chars, n);
628 }

632 template<typename CharT, typename Traits>
633 void insert_aligned(std::basic_ostream<CharT, Traits>& os, const basic_string_view<CharT, Traits>& str)
634 {
635  const std::size_t size = str.size();
636  const std::size_t alignment_size = static_cast< std::size_t >(os.width()) - size;
637  const bool align_left = (os.flags() & std::basic_ostream<CharT, Traits>::adjustfield) == std::basic_ostream<CharT, Traits>::left;
638  if (!align_left)
639  {
640  detail::insert_fill_chars(os, alignment_size);
641  if (os.good())
642  os.write(str.data(), size);
643  }
644  else
645  {
646  os.write(str.data(), size);
647  if (os.good())
648  detail::insert_fill_chars(os, alignment_size);
649  }
650 }
