96  // this 'as_std_shared_ptr' policy has been selected, because:
97  // * it has automatic memory management; should not leak anything
98  // * we can copy the object inside the variant, without invoking the copy constructor (could be disabled by user)

179  "Only one set of 'default_arguments' can be provided during registration of a constructor!");
180 
181  static_assert((count_param_names<Args...>::value <= 1),
182  "Only one set of 'parameter_names' can be provided during registration of a constructor!");
183 
184  static_assert(((!has_param_names<Args...>::value) ||
185  (param_names_count<Args...>::value == sizeof...(Ctor_Args))),
186  "The provided amount of names in 'parameter_names' does not match argument count of the constructor signature.");
187 
188  // when no policy was added, we need a default policy
189  using policy_list = conditional_t< type_list_size<policy_types_found>::value == 0,
190  default_create_policy,
191  policy_types_found>;
192 
193  // at the moment we only supported one policy
194  using first_prop_policy = typename std::tuple_element<0, as_std_tuple_t<policy_list>>::type;
195  using metadata_count = count_type<::rttr::detail::metadata, type_list<Args...>>;
196  m_ctor = create_constructor_wrapper<first_prop_policy>(std::move(get_metadata(std::forward<Args>(args)...)),
197  std::move(get_default_args<type_list<Ctor_Args...>, constructor_type>(std::forward<Args>(args)...)),
198  std::move(create_param_infos<type_list<Ctor_Args...>, constructor_type>(std::forward<Args>(args)...)));
199  return registration::class_<Class_Type>(m_reg_exec);
200  }
