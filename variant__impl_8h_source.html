237 template<typename T>
238 RTTR_INLINE bool variant::convert(T& value) const
239 {
240  bool ok = false;
241 
242  const type source_type = get_type();
243  const type target_type = type::get<T>();
244  if (target_type == source_type)
245  {
246  value = const_cast<variant&>(*this).get_value<T>();
247  ok = true;
248  }
249  else if(try_basic_type_conversion(value))
250  {
251  ok = true;
252  }
253  else if (const auto& converter = source_type.get_type_converter(target_type))
254  {
255  const auto target_converter = static_cast<const detail::type_converter_target<T>*>(converter);
256  value = target_converter->convert(get_ptr(), ok);
257  }
258  else if (target_type == type::get<std::nullptr_t>())
259  {
260  if (is_nullptr())
261  ok = ptr_to_nullptr(value);
262  }
263  else
264  {
265  ok = try_pointer_conversion(value, source_type, target_type);
266  }
267 
268  return ok;
269 }
