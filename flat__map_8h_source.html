112  void insert(const Key&& key, Value&& value)
113  {
114  if (find(key) != end())

117  m_key_list.push_back(key_data_type{std::move(key), has_type()(key)});
118  std::stable_sort(m_key_list.begin(), m_key_list.end(), typename key_data_type::order());
