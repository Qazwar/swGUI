338  static inline std::string gen_unique_filename(const std::string& filename) {
339  std::string::size_type pos_dot = filename.find_last_of(".");
340  const std::string file_basename = filename.substr(0, pos_dot);
341  const std::string file_suffix = filename.substr(pos_dot + 1);

344  int counter = 1;
345  while (file_exists(new_path)) {
346  new_path = new_path = file_basename + "_" + std::to_string(counter) + "." + file_suffix;
347  ++counter;
348  }
349 
350  return new_path;
351  }
