36 class property_wrapper<member_func_ptr, Getter, Setter, Acc_Level, return_as_copy, set_value, Metadata_Count>
37  : public property_wrapper_base, public metadata_handler<Metadata_Count>
38 {
39  using return_type = typename function_traits<Getter>::return_type;
40  using arg_type = typename param_types<Setter, 0>::type;

42

45  Getter get, Setter set,
46  std::array<metadata, Metadata_Count> metadata_list) RTTR_NOEXCEPT
47  : property_wrapper_base(name, declaring_type),
48  metadata_handler<Metadata_Count>(std::move(metadata_list)),
49  m_getter(get), m_setter(set)
50  {
51  static_assert(function_traits<Getter>::arg_count == 0, "Invalid number of argument, please provide a getter-member-function without arguments.");
52  static_assert(function_traits<Setter>::arg_count == 1, "Invalid number of argument, please provide a setter-member-function with exactly one argument.");
53  static_assert(std::is_same<return_type, arg_type>::value, "Please provide the same signature for getter and setter!");
54 
55  init();
56  }
