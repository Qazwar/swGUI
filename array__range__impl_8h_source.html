252 template<typename T, typename Predicate>
253 RTTR_INLINE bool array_range<T, Predicate>::empty() const
254 {
255  if (m_begin == m_end)
256  return true;
257 
258  const_iterator itr{m_begin, this};
259  if (m_pred(*itr))
260  return false;
261 
262  ++itr.m_ptr;
263  while(itr.m_ptr != m_end && !m_pred(*itr.m_ptr))
264  {
265  ++itr.m_ptr;
266  }
267 
268  return (itr == cend());
269 }
