497 RTTR_INLINE static std::size_t generate_hash(const char* text, std::size_t length)
498 {
499  const std::size_t magic_prime = static_cast<std::size_t>(0x01000193);
500  std::size_t hash = static_cast<std::size_t>(0xcbf29ce4);
501 
502  for (std::size_t i = 0; i < length; ++i)
503  hash = (hash ^ text[i]) * magic_prime;
504 
505  return hash;
506 }
